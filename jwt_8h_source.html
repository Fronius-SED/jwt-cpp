375  inline std::string extract_pubkey_from_cert(const std::string& certstr, const std::string& pw,
376  std::error_code& ec) {
377  ec.clear();
378 #if OPENSSL_VERSION_NUMBER <= 0x10100003L
379  std::unique_ptr<BIO, decltype(&BIO_free_all)> certbio(
380  BIO_new_mem_buf(const_cast<char*>(certstr.data()), static_cast<int>(certstr.size())), BIO_free_all);
381 #else
382  std::unique_ptr<BIO, decltype(&BIO_free_all)> certbio(
383  BIO_new_mem_buf(certstr.data(), static_cast<int>(certstr.size())), BIO_free_all);
384 #endif

389  }

392  PEM_read_bio_X509(certbio.get(), nullptr, nullptr, const_cast<char*>(pw.c_str())), X509_free);

401  }
